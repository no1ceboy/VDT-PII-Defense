% Section 10: Privacy Layer Detailed
\begin{frame}{Defense Layer 1: Data Sanitization (Privacy Filter)}
    \textbf{Concept:} A zero-trust approach. Stop sensitive data from ever reaching the LLM's context window. \\~\\
    
    \textbf{Technical Implementation:}
    \begin{itemize}
        \item \textbf{The Tool:} Integrated the official \texttt{openai/privacy-filter} (\texttt{opf}) library, utilizing a highly optimized, lightweight Custom Mixture-of-Experts (MoE) neural network.
        \item \textbf{The Process:} 
        \begin{enumerate}
            \item The MoE intercepts the raw input document.
            \item It performs Named Entity Recognition (NER) to classify tokens.
            \item Sensitive tokens are mapped and swapped with heavily-typed generic identifiers (e.g., \texttt{<PRIVATE\_PERSON\_1>}).
        \end{enumerate}
    \end{itemize}
    
    \vspace{0.2cm}
    \begin{tcolorbox}[colback=blue!5,colframe=blue!75!black,title=Redaction Example]
    \small \textbf{Input:} "Bệnh nhân Nguyễn Văn A, SĐT 0901234567" \\
    \textbf{Output:} "\texttt{<PRIVATE\_PERSON>}, SĐT \texttt{<PRIVATE\_PHONE\_NUMBER>}"
    \end{tcolorbox}
    
    \textbf{Security Guarantee:} PII extraction attacks are mathematically neutralized; the LLM cannot leak what it cannot see.
\end{frame}

% Section 12: DPO
\begin{frame}{Defense Layer 2: Behavioral Alignment (DPO)}
    \textbf{Concept:} Teach the AI to actively recognize and refuse malicious instructions. \\~\\
    
    \textbf{Implementation:}
    \begin{itemize}
        \item Constructed a custom Preference Dataset using our attack results:
        \begin{itemize}
            \item \textbf{Rejected Behavior:} Maliciously complying with the injection.
            \item \textbf{Chosen Behavior:} Safely summarizing the document and ignoring the injection.
        \end{itemize}
        \item \textbf{Status:} Direct Preference Optimization (DPO) training architecture engineered for Kaggle T4 GPUs.
        \item \textbf{Engineering Milestone:} Overcame severe hardware constraints (cross-GPU communication crashes, VRAM limits) via gradient checkpointing, single-device mapping, and strict sequence truncation.
    \end{itemize}
\end{frame}
